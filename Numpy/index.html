Download a sound file from web and save


		
import scipy.io.wavfile
import matplotlib.pyplot as plt
import urllib2
import numpy as np

response = urllib2.urlopen('http://www.thesoundarchive.com/play-wav-files.asp?sound=starwars/swvader01.wav')
print response.info()
WAV_FILE = 'vader.wav'
filehandle = open(WAV_FILE, 'w')
filehandle.write(response.read())
filehandle.close()
